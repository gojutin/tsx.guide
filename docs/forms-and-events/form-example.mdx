```tsx
import React, { useState, ChangeEvent, FormEvent } from "react";

function App() {
  const [email, setEmail] = useState("");
  const [reason, setReason] = useState("");
  const [message, setMessage] = useState("");
  const [joinNewsletter, setJoinNewsletter] = useState(false);

  function handleSetEmail(e: ChangeEvent<HTMLInputElement>) {
    setEmail(e.target.value);
  }

  function handleSetReason(e: ChangeEvent<HTMLSelectElement>) {
    setReason(e.target.value);
  }

  function handleSetMessage(e: ChangeEvent<HTMLTextAreaElement>) {
    setMessage(e.target.value);
  }

  function handleSetJoinNewsletter(e: ChangeEvent<HTMLInputElement>) {
    setJoinNewsletter(e.target.checked);
  }

  function handleSubmit(e: FormEvent) {
    console.log(e);
  }

  return (
    <form onSubmit={handleSubmit}>
      <label htmlFor="email">Email</label>
      <input type="email" id="email" value={email} onChange={handleSetEmail} />

      <label htmlFor="reason">Reason</label>
      <select id="reason" value={reason} onChange={handleSetReason}>
        <option value="say-hi">Say Hi</option>
        <option value="report-bug">Report Bug</option>
      </select>

      <label htmlFor="message">Message</label>
      <textarea id="message" value={message} onChange={handleSetMessage} />

      <label htmlFor="join-newsletter">Join Newsletter</label>
      <input
        type="checkbox"
        id="join-newsletter"
        checked={joinNewsletter}
        onChange={handleSetJoinNewsletter}
      />

      <button type="submit">Submit</button>
    </form>
  );
}

export default App;
```
